export function InteractiveChain() {
  const [active, setActive] = useState(0);
  const steps = [
    { label: '.styles()', layer: 'base', repeatable: false, available: 'Entry point',
      description: 'Entry point. Define base CSS that applies unconditionally to every instance.',
      code: `ds.styles({
  display: 'inline-flex',
  alignItems: 'center',
  fontWeight: 500,
})` },
    { label: '.variant()', layer: 'variants', repeatable: true, available: '.styles()',
      description: 'Named variant axes. Call multiple times for size, color, shape — each gets its own axis.',
      code: `.variant({
  prop: 'size',
  variants: {
    sm: { px: 10, fontSize: 13 },
    md: { px: 16, fontSize: 14 },
    lg: { px: 24, fontSize: 16 },
  },
})` },
    { label: '.compound()', layer: 'compounds', repeatable: true, available: '.variant()',
      description: 'Conditional styles when multiple variants match. Order-sensitive — later compounds win.',
      code: `.compound(
  { size: 'lg', intent: 'primary' },
  { boxShadow: '0 2px 8px rgba(0,0,0,0.2)' }
)` },
    { label: '.states()', layer: 'states', repeatable: false, available: '.compound()',
      description: 'Boolean state toggles via data-attributes. Maps to @layer states.',
      code: `.states({
  loading: { opacity: '0.5', pointerEvents: 'none' },
  elevated: { boxShadow: 'glow-accent' },
})` },
    { label: '.system()', layer: 'system', repeatable: false, available: '.states()',
      description: 'Bind to design system token groups — spacing, colors, typography.',
      code: `.system({
  space: true,
  color: true,
  typography: true,
})` },
    { label: '.props()', layer: 'custom', repeatable: false, available: '.system()',
      description: 'Custom props resolved at render time. Ends the chain before terminal.',
      code: `.props({
  glow: (v) =>
    \`box-shadow: 0 0 \${v}px currentColor\`,
})` },
  ];
  return <ChainStep steps={steps} activeStep={active} onStepClick={setActive} />;
}

## Callout Variants

<Callout variant="info" title="Cross-reference">
  The cascade contract is documented in the Architecture section. Each builder method maps to a specific `@layer` — see the chain visualization above.
</Callout>

<Callout variant="tip" title="Best practice">
  Define your color scheme as a variant that rebinds `--color-scheme-*` weights. Style variants then reference `scheme.500`, `scheme.400` etc. — the component never knows which palette is active.
</Callout>

<Callout variant="warn" title="Breaking in v2.3">
  The `.states()` map signature changed from state names to CSS selectors as keys. See the migration guide for details on updating existing components.
</Callout>

<Callout variant="danger" title="Do not">
  Never add React resolve aliases to `vite.config.ts`. They silently break the extraction transform pipeline. If you need to alias a package, use the `resolve.alias` field in the Vite config with extreme caution.
</Callout>

### Collapsible

<SyntaxBlock title="full-system-setup.ts" language="tsx" showLineNumbers collapsible>{`import { createSystem, createTheme, createTransform } from '@animus-ui/system';
import { space, typography, color, border, shadows } from '@animus-ui/system/groups';

const fluid = createTransform('fluid', (value) => {
  const [min, max] = String(value).split('-').map(Number);
  const minRem = min / 16;
  const maxRem = max / 16;
  const slope = ((max - min) / 880) * 100;
  const intercept = minRem - (slope / 100) * (480 / 16);
  return \`clamp(\${minRem}rem, \${intercept}rem + \${slope}vw, \${maxRem}rem)\`;
});

export const tokens = createTheme()
  .addBreakpoints({ sm: 768, md: 1024, lg: 1200 })
  .addColors({
    gray: { 50: '#fafafa', 900: '#111' },
    fire: { 500: '#FF2800', 700: '#C1121F' },
  })
  .addScale({ name: 'space', values: { 4: '0.25rem', 8: '0.5rem', 16: '1rem' } })
  .build();

export const { system: ds } = createSystem()
  .addGroup('surface', { ...color, ...border, ...shadows })
  .addGroup('text', { ...typography })
  .addGroup('space', { ...space })
  .build();`}</SyntaxBlock>
